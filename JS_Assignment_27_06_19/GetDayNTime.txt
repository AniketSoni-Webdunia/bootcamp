
function GetDayNTime(){

var dayArray=["Sunday","Mnday","Tuesday","Wednesday","Thrusday","Friday","Saturday"];

var current_date = new Date();

dv = current_date.getDay();

//time

var ap="AM"

if(h>12){
   h=h-12;
   ap="PM"
}
console.log(h+""+ap+" : "+m+" "+s);

}

GetDayNTime();
