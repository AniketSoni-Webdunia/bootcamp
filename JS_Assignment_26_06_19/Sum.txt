let sum=0;

function AskForNumber(n){
    if(n==0)
       return 1;
    else{
       let num = +prompt("Enter a number","");
       try{
       if(isNaN(num)){
           throw new Error("Invalid Input");
          }
       sum+=num;   
       AskForNumber(n-1);
       }catch(e){
            alert(e);
            
       }
    }   
}
